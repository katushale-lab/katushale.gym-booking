function doGet(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('все_записи'); if (e.parameter.action === 'read') { const data = sheet.getDataRange().getValues(); return ContentService.createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } if (e.parameter.action === 'add') { const day = e.parameter.day; const timeSlot = e.parameter.time; const hall = e.parameter.hall; const trainer = e.parameter.trainer; const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] === day && data[i][1] === timeSlot && data[i][2] === hall) { if (trainer === '') { sheet.deleteRow(i + 1); } else { sheet.getRange(i + 1, 4).setValue(trainer); } return ContentService.createTextOutput('OK').setMimeType(ContentService.MimeType.TEXT); } } if (trainer !== '') { sheet.appendRow([day, timeSlot, hall, trainer]); } return ContentService.createTextOutput('OK').setMimeType(ContentService.MimeType.TEXT); } if (e.parameter.action === 'clear') { const lastRow = sheet.getLastRow(); if (lastRow > 1) { sheet.getRange(2, 1, lastRow - 1, 4).clearContent(); } return ContentService.createTextOutput('OK').setMimeType(ContentService.MimeType.TEXT); } // По умолчанию — возвращаем все данные const data = sheet.getDataRange().getValues(); return ContentService.createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
